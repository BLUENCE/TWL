def calc_sum(numbs):
    result = 0
    for num in numbs:
        result = result + num
    return result

def calc_len(numbs):
    result = 0
    for num in  numbs:
        result = result + 1
    return result

def calc_average(numbs):
    total = calc_sum(numbs)
    length = calc_len(numbs)
    return total / length

scores = [50, 60, 70]
total = calc_sum(scores)
print(total)
